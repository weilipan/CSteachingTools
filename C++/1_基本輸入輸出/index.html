C++ 基本輸入輸出與變數記憶體視覺化

逐步執行程式碼，觀察變數宣告、記憶體配置與 cin/cout 的運作。

程式碼 (Code)


#include <iostream>
#include <string>
using namespace std;

int main() {
    // 1. 變數宣告與初始化
    int age = 18;
    double height = 1.75;
    string name = "Alice";

    // 2. 輸出 (cout)
    cout << "Name: " << name << endl;
    cout << "Age: " << age << endl;

    // 3. 輸入 (cin)
    int newAge;
    cout << "Enter new age: ";
    cin >> newAge;

    // 4. 更新變數
    age = newAge;
    cout << "Updated Age: " << age << endl;

    return 0;
}
