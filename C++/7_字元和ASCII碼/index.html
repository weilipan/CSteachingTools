main.cpp 

#include <iostream>

int main() {
    // Character declaration
    char c = 'A';
    
    // Implicit casting to integer (ASCII value)
    int ascii_val = c; 
    
    std::cout << "Char: " << c << std::endl;      // Output: A
    std::cout << "ASCII: " << ascii_val << std::endl; // Output: 65

    // Math with characters
    // 'A' + 1 = 66 ('B')
    char next = c + 1;
    
    return 0;
}
