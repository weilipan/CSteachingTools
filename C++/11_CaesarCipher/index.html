#include <iostream>
#include <string>
#include <cctype>
using namespace std;

// 加密單一字元
char encryptChar(char c, int key) {
    if (isupper(c))
        return (c - 'A' + key) % 26 + 'A';
    if (islower(c))
        return (c - 'a' + key) % 26 + 'a';
    return c; // 標點/空白原封不動
}

// 解密單一字元（負數解法）
char decryptChar(char c, int key) {
    if (isupper(c))
        return (c - 'A' + 26 - key) % 26 + 'A';
    if (islower(c))
        return (c - 'a' + 26 - key) % 26 + 'a';
    return c;
}

// 走訪字串，組裝結果
string caesar(string text, int key, bool encrypt) {
    string result = "";
    for (char c : text) {
        if (encrypt) result += encryptChar(c, key);
        else        result += decryptChar(c, key);
    }
    return result;
}

int main() {
    string plain = "Hello, World!";
    int key = 3;

    string cipher  = caesar(plain,  key, true);
    string decoded = caesar(cipher, key, false);

    cout << "明文:  " << plain   << endl;
    cout << "密文:  " << cipher  << endl;
    cout << "解密:  " << decoded << endl;
    return 0;
}
